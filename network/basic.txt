# Create a simulator object
set ns [new Simulator]

# Create two nodes
set n0 [$ns node]
set n1 [$ns node]

# Create a duplex link between n0 and n1 with 1Mb bandwidth and 10ms delay
$ns duplex-link $n0 $n1 1Mb 10ms DropTail

# Run the simulation
$ns run
