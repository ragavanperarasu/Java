        scanf("%d", &bt[i]);
        rem[i] = bt[i];
    }
    printf("Enter Time Quantum: ");
    scanf("%d", &tq);

    printf("\n--- Round Robin ---\n");
    printf("PID\tAT\tBT\tCT\tTAT\tWT\n");

    while (done < n) {
        bool idle = true;
        for (i = 0; i < n; i++) {
            if (rem[i] > 0 && at[i] <= time) {
                idle = false;
                int run = (rem[i] < tq) ? rem[i] : tq;
                rem[i] -= run;
                time += run;
                if (rem[i] == 0) {
                    int ct = time;
                    int tat = ct - at[i];
                    int wt = tat - bt[i];
                    total_tat += tat; total_wt += wt;
                    printf("P%d\t%d\t%d\t%d\t%d\t%d\n", pid[i], at[i], bt[i], ct, tat, wt);
                    done++;
                }
            }
        }
        if (idle) time++;
    }

    printf("\nAverage TAT = %.2f", total_tat / n);
