#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>
#include <unistd.h>

#define N 5           
#define EAT_TIMES 3   

sem_t chopstick[N];   
sem_t waiter;         

void* philosopher(void* num) {
    int id = *(int*)num;

    for (int i = 0; i < EAT_TIMES; i++) {
        printf("Philosopher %d is thinking \n", id);
        sleep(1);

        sem_wait(&waiter);

        sem_wait(&chopstick[id]);
        sem_wait(&chopstick[(id + 1) % N]);

       
        printf("Philosopher %d is eating  (round %d)\n", id, i + 1);
        sleep(2);

       
        sem_post(&chopstick[id]);
        sem_post(&chopstick[(id + 1) % N]);

       
        sem_post(&waiter);

        printf("Philosopher %d finished eating and is thinking again \n", id);
        sleep(1);
    }

    printf("Philosopher %d has finished all meals \n", id);
    return NULL;
}

int main() {
    pthread_t tid[N];
    int phil_id[N];

    
    sem_init(&waiter, 0, N - 1);  // Allow N-1 philosophers to try eating at once
    for (int i = 0; i < N; i++) {
        sem_init(&chopstick[i], 0, 1);
        phil_id[i] = i;
    }

   
    for (int i = 0; i < N; i++) {
        pthread_create(&tid[i], NULL, philosopher, &phil_id[i]);
    }

    
    for (int i = 0; i < N; i++) {
        pthread_join(tid[i], NULL);
    }

    sem_destroy(&waiter);
    for (int i = 0; i < N; i++)
        sem_destroy(&chopstick[i]);

    printf("All philosophers have finished eating \n");
    return 0;
}
